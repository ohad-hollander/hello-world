FIXED-TIMESTEP GAME LOOP (FOUND-01):
  const FIXED_STEP = 1 / 60;   // 60 Hz update rate
  let accumulator = 0;
  let lastTime = null;
  let frameCount = 0;
  let fpsTimer = 0;

  function update(dt) {
    // Phase 1: no entities yet — loop proof only
  }

  function render() {
    ctx.fillStyle = '#000';
    ctx.fillRect(0, 0, 672, 768);
    ctx.fillStyle = '#0f0';
    ctx.fillRect(316, 380, 40, 40); // placeholder — removed in Phase 2
  }

  function loop(timestamp) {
    if (lastTime === null) lastTime = timestamp;
    let dt = (timestamp - lastTime) / 1000;
    lastTime = timestamp;

    if (dt > 0.25) dt = 0.25; // clamp: survive tab switches without spiral of death

    accumulator += dt;
    while (accumulator >= FIXED_STEP) {
      update(FIXED_STEP);
      accumulator -= FIXED_STEP;
    }

    render();

    frameCount++;
    fpsTimer += dt;
    if (fpsTimer >= 1.0) {
      console.log('FPS: ' + frameCount + ' | keys:', JSON.stringify(keys));
      frameCount = 0;
      fpsTimer = 0;
    }

    requestAnimationFrame(loop);
  }

  requestAnimationFrame(loop);

Key implementation notes:
- dt clamped to 0.25s — prevents spiral of death on tab switch (noted in STATE.md concerns)
- lastTime = null sentinel — avoids a huge first-frame dt spike
- keys object logged with FPS — proves FOUND-03 readable per-frame (ROADMAP success criterion)
- FIXED_STEP is a constant — do NOT derive from screen refresh rate
- update() is intentionally empty in Phase 1 — Phase 2 adds entity updates
